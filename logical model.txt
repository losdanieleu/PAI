 *id: serial
 --
 *name: text

 *likes: unsigned bigint(20)
}

 *id: serial
 --
 *name: text
 *password: text
 *created_at: timestamp

}

 *id: serial
 --
 *created_at: timestamp
 *last_modified: timestamp

}

 *id: serial
 --

}
